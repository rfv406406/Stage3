docker ps
	// 查詢 docker 執行中的容器

docker ps -a
	// 能顯示沒有 run 的 container
	// 取得 ContainerID

docker stop ContainerID
	// 停止該容器運行

docker cp .env ContainerID:/app/
	// 複製 .env 檔案到 docker container 中:
	// cp 複製，覆蓋優先
	// 防覆蓋 用 cp -i 會有選擇選項
	// app 是我 設定的檔案路徑

docker start ContainerID
	// 啟用指定的 docker container

docker images
	// 列出所有 Image

docker rm ContainerID
	// 刪除指定 container
